        }

        function r() {
            var t = e("link").filter(function(e) {
                    return "stylesheet" === e.getAttribute("rel") && e !== n
                }),

                o = t.concat(r);
            o.forEach(function(e) {
                e.parentElement.removeChild(e)
            })
        }
        var n = window.document.createElement("link");
